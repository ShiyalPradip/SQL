1.NAME OF THE EMPLOYEE AND HIS LOCATION OF ALL THE EMPLOYEES. 
    SELECT ENAME, LOC FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO;

2.WAQTD DNAME AND SALARY FOR ALL THE EMPLOYEE WORKING IN ACCOUNTING. 
    SELECT DNAME, SAL FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE DEPT.DNAME = 'ACCOUNTING';

3.WAQTD DNAME AND ANNUAL SALARY FOR ALL EMPLOYEES WHOS SALARY IS MORE THAN 2340 
    SELECT DNAME, (SAL * 12) AS "Annual Salary" FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE SAL > 2340;

4.WAQTD ENAME AND DNAME FOR EMPLOYEES HAVING CHARACTER 'A' IN THEIR DNAME  
    SELECT ENAME, DNAME FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE DEPT.DNAME LIKE '%A%';

5.WAQTD ENAME AND DNAME FOR ALL THE EMPLOYEES WORKING AS SALESMAN 
    SELECT ENAME, DNAME FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE EMP.JOB = 'SALESMAN';

6.WADTD DNAME AND JOB FOR ALL THE EMPLOYEES WHOS JOB AND DNAME STARTS WITH CHARACTER 'S' 
    SELECT DNAME, JOB FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE EMP.JOB LIKE 'S%' AND DEPT.DNAME LIKE 'S%';

7.WAQTD DNAME AND MGR NO FOR EMPLOYEES REPORTING TO 7839 
    SELECT DNAME, MGR FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE EMP.MGR = 7839;

8.WAQTD DNAME AND HIREDATE FOR EMPLOYEES HIRED AFTER 83 INTO ACCOUNTING OR RESEARCH DEPT  
    SELECT DNAME, HIREDATE FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE HIREDATE > '1983-01-01' AND DNAME IN ('ACCOUNTING', 'RESEARCH');

9.WAQTD ENAME AND DNAME OF THE EMPLOYEES WHO ARE GETTING COMM IN DEPT 10 OR 30  
    SELECT ENAME, DNAME FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE COMM IS NOT NULL AND EMP.DEPTNO IN (10,30);

10.WAQTD DNAME AND EMPNO FOR ALL THE EMPLOYEES WHO'S EMPNO ARE (7839,7902) AND ARE WORKING IN LOC NEW YORK. 
    SELECT DNAME, EMPNO FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE EMPNO IN (7839, 7902) AND LOC = 'NEW YORK';

11.WAQTD LOC AND AVERAGE SALARY GIVEN FOR EACH LOCATION BY EXCLUDING ALL THE EMPLOYEES WHOS SECOND CHAR IS A IN THEIR NAME . 
    SELECT LOC, AVG(SAL) AS 'AVERAGE SALARY' FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE EMP.ENAME NOT LIKE '_A%' GROUP BY DEPT.LOC;

12.WAQTD NAME OF THE EMP AND HIS LOC IF EMPLOYEE IS WORKING AS MANAGER AND WORKING UNDER THE EMPLOYEE WHOS EMPNO IS 7839 
    SELECT EMP.ENAME, DEPT.LOC FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE EMP.JOB = 'MANAGER' AND EMP.MGR = 7839;

13.WAQTD DNAME AND EMPLOYEE ID'S OF ALL THE EMPLOYEES WHO ARE CLERKS AND HAVING REPORTING MANAGERS. 
    SELECT DNAME, EMPNO FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO WHERE JOB = 'CLERK' AND MGR IS NOT NULL;

14.WAQTD DNAME AND TOTAL SALARY GIVEN TO THAT DEPT IF THERE ARE ATLEAST 4 EMPLOYEES WORKING FOR EACH DEPT. 
    SELECT DNAME, SUM(SAL) AS TOTAL_SALARY FROM EMP INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO GROUP BY DNAME HAVING COUNT(EMPNO) >= 4;
